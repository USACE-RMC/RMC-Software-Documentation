This guide provides a direct mapping between Flask/Python concepts and their ASP.NET Core/C# equivalents. Use this as a reference when reading or writing ASP.NET code.

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Authorization;
using SystemResponse.Toolboxes.BEPProgression.Schmertmann;

namespace RMCToolboxes.API.Controllers;

[ApiController]  // "This class handles API requests"
[Route("api/toolboxes/bep-progression/schmertmann")]
[Authorize]  // "User must be logged in"
public class SchmertmannController : ControllerBase
{
    private readonly ToolboxDbContext _db;
    private readonly ILogger<SchmertmannController> _logger;

    public SchmertmannController(
        ToolboxDbContext db,
        ILogger<SchmertmannController> logger)
    {
        _db = db;
        _logger = logger;
    }

    [HttpPost("calculate/{analysisId}")]
    public async Task<IActionResult> Calculate(int analysisId)
    {
        // 1. Get user from JWT token (automatic)
        var userId = User.FindFirstValue(
            ClaimTypes.NameIdentifier);

        // 2. Load from database
        var analysis = await _db.Analyses.FindAsync(analysisId);

        // 3. Check ownership
        if (analysis.UserId != userId)
            return Forbid();  // HTTP 403

        // 4. Extract parameters from database
        var config = analysis.Parameters;  // JSONB field

        // 5. Run calculation
        var model = new SchmertmannModel(config);
        var result = await model.RunAsync();

        // 6. Save results back to database
        analysis.Results = result;
        await _db.SaveChangesAsync();

        // 7. Return response
        return Ok(new {
            status = "success",
            inputs = new List<string>(),
            warnings = new List<string>(),
            errors = new List<string>(),
            can_run = true,
            results = result,
            results_cleared = false
        });
    }
}`}
        </pre>
      ),

## Key Syntax Differences

### Decorators vs Attributes

Flask uses Python decorators. ASP.NET uses C# attributes (which look similar but use square brackets).

## Async/Await

Both Flask and ASP.NET support async operations, but ASP.NET uses it more pervasively.

<DocTabs
  items={[

{`# Synchronous (common)
def calculate(analysis_id):
    analysis = Analysis.query.get(analysis_id)
    result = model.run()
    return jsonify(result)

    analysis = await Analysis.query.get(analysis_id)
    result = await model.run_async()

public async Task<IActionResult> Calculate(int analysisId)
{
    // await for database operations
    var analysis = await _db.Analyses.FindAsync(analysisId);

    // await for long-running calculations
    var result = await model.RunAsync();
