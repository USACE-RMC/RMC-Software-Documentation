## Controller Template

### Basic CRUD Controller

```csharp
[ApiController]
[Route("api/[controller]")]
[Authorize]
public class AnalysesController : ControllerBase
{
    private readonly AppDbContext _db;

    public AnalysesController(AppDbContext db) => _db = db;

    // GET /api/analyses
    [HttpGet]
    public async Task<IActionResult> GetAll()
    {
        var userId = GetCurrentUserId();
        var items = await _db.Analyses
            .Where(a => a.UserId == userId)
            .ToListAsync();
        return Ok(items);
    }

    // GET /api/analyses/5
    [HttpGet("{id}")]
    public async Task<IActionResult> GetById(int id)
    {
        var item = await _db.Analyses.FindAsync(id);
        if (item == null) return NotFound();
        if (item.UserId != GetCurrentUserId()) return Forbid();
        return Ok(item);
    }

    // POST /api/analyses
    [HttpPost]
    public async Task<IActionResult> Create([FromBody] AnalysisDto dto)
    {
        var analysis = new Analysis
        {
            Name = dto.Name,
            UserId = GetCurrentUserId()
        };
        _db.Analyses.Add(analysis);
        await _db.SaveChangesAsync();
        return Created($"/api/analyses/{analysis.Id}", analysis);
    }

    // PUT /api/analyses/5
    [HttpPut("{id}")]
    public async Task<IActionResult> Update(int id, [FromBody] AnalysisDto dto)
    {
        var analysis = await _db.Analyses.FindAsync(id);
        if (analysis == null) return NotFound();
        if (analysis.UserId != GetCurrentUserId()) return Forbid();

        analysis.Name = dto.Name;
        analysis.UpdatedAt = DateTime.UtcNow;
        await _db.SaveChangesAsync();

        return Ok(analysis);
    }

    // DELETE /api/analyses/5
    [HttpDelete("{id}")]
    public async Task<IActionResult> Delete(int id)
    {
        var analysis = await _db.Analyses.FindAsync(id);
        if (analysis == null) return NotFound();
        if (analysis.UserId != GetCurrentUserId()) return Forbid();

        _db.Analyses.Remove(analysis);
        await _db.SaveChangesAsync();

        return NoContent();
    }

    private string GetCurrentUserId() =>
        User.FindFirstValue(ClaimTypes.NameIdentifier)!;
}
```

### Calculation Endpoint

```csharp
[HttpPost("calculate/{analysisId}")]
public async Task<IActionResult> Calculate(int analysisId)
{
    // 1. Get current user
    var userId = GetCurrentUserId();

    // 2. Load analysis
    var analysis = await _db.Analyses.FindAsync(analysisId);
    if (analysis == null) return NotFound();

    // 3. Check authorization
    if (analysis.UserId != userId) return Forbid();

    // 4. Run calculation (library call)
    var config = MapToConfig(analysis.Parameters);
    var model = new CalculationModel(config);
    var result = await model.RunAsync();

    // 5. Save results
    analysis.Results = JsonSerializer.SerializeToDocument(result);
    analysis.UpdatedAt = DateTime.UtcNow;
    await _db.SaveChangesAsync();

    // 6. Return messaging contract response
    return Ok(new CalculationResponse
    {
        Status = "success",
        Results = result,
        Warnings = result.Warnings,
        CanRun = true
    });
}
```

## Entity Template

```csharp
public class Analysis
{
    public int Id { get; set; }

    [Required]
    [MaxLength(255)]
    public string Name { get; set; } = string.Empty;

    [Required]
    public string UserId { get; set; } = string.Empty;

    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;

    public DateTime UpdatedAt { get; set; } = DateTime.UtcNow;

    // JSONB column for flexible parameters
    [Column(TypeName = "jsonb")]
    public JsonDocument? Parameters { get; set; }

    // JSONB column for calculation results
    [Column(TypeName = "jsonb")]
    public JsonDocument? Results { get; set; }

    // Navigation property
    public User User { get; set; } = null!;
}
```

## Logging Patterns

```csharp
public class AnalysisController : ControllerBase
{
    private readonly ILogger<AnalysisController> _logger;

    public AnalysisController(ILogger<AnalysisController> logger)
    {
        _logger = logger;
    }

    [HttpPost("calculate/{id}")]
    public async Task<IActionResult> Calculate(int id)
    {
        // Use structured logging with named parameters
        _logger.LogInformation(
            "[API] POST /api/analysis/calculate/{AnalysisId}",
            id);

        try
        {
            _logger.LogDebug(
                "[CALC] Starting calculation: Id={Id}, Iterations={Iterations}",
                id, config.Iterations);

            // ... calculation ...

            _logger.LogInformation(
                "[CALC] Completed in {ElapsedMs}ms",
                stopwatch.ElapsedMilliseconds);

            return Ok(result);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex,
                "[ERROR] Calculation failed for analysis {Id}",

            throw;
        }
    }
}
```

## Common Gotchas

### Forgetting await

```csharp
// WRONG: Returns Task instead of data
public async Task<IActionResult> Get()
{
    var items = _db.Analyses.ToListAsync(); // Missing await!
    return Ok(items);
}

// CORRECT
public async Task<IActionResult> Get()
{
    var items = await _db.Analyses.ToListAsync();
    return Ok(items);
}
```

### Forgetting null checks

```csharp
// WRONG: NullReferenceException if not found
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    return Ok(item.Name); // Crashes if item is null!
}

// CORRECT
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    if (item == null) return NotFound();
    return Ok(item.Name);
}
```

### Forgetting to save

```csharp
// WRONG: Changes not persisted
public async Task<IActionResult> Update(int id, UpdateDto dto)
{
    var item = await _db.Analyses.FindAsync(id);
    item.Name = dto.Name;
    // Missing SaveChangesAsync!
    return Ok(item);
}

// CORRECT
public async Task<IActionResult> Update(int id, UpdateDto dto)
{
    var item = await _db.Analyses.FindAsync(id);
    item.Name = dto.Name;
    await _db.SaveChangesAsync();
    return Ok(item);
}
```

### Forgetting authorization

```csharp
// WRONG: Any user can access any analysis
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    return Ok(item);
}

// CORRECT: Check ownership
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    if (item == null) return NotFound();
    if (item.UserId != GetCurrentUserId()) return Forbid();
    return Ok(item);
}
```
