## Common Controller Patterns

### Basic CRUD Controller

```csharp
[ApiController]
[Route("api/[controller]")]
[Authorize]
public class AnalysesController : ControllerBase
{
    private readonly ToolboxDbContext _db;

    public AnalysesController(ToolboxDbContext db) => _db = db;

    // GET /api/analyses
    [HttpGet]
    public async Task<IActionResult> GetAll()
    {
        var items = await _db.Analyses.ToListAsync();
        return Ok(items);
    }

    // GET /api/analyses/5
    [HttpGet("{id}")]
    public async Task<IActionResult> GetById(int id)
    {
        var item = await _db.Analyses.FindAsync(id);
        return item == null ? NotFound() : Ok(item);
    }

    // POST /api/analyses
    [HttpPost]
    public async Task<IActionResult> Create([FromBody] AnalysisDto dto)
    {
        var analysis = new Analysis { Name = dto.Name };
        _db.Analyses.Add(analysis);
        await _db.SaveChangesAsync();
        return Created($"/api/analyses/{analysis.Id}", analysis);
    }

    // PUT /api/analyses/5
    [HttpPut("{id}")]
    public async Task<IActionResult> Update(int id, [FromBody] AnalysisDto dto)
    {
        var analysis = await _db.Analyses.FindAsync(id);
        if (analysis == null) return NotFound();
        analysis.Name = dto.Name;
        await _db.SaveChangesAsync();
        return Ok(analysis);
    }

    // DELETE /api/analyses/5
    [HttpDelete("{id}")]
    public async Task<IActionResult> Delete(int id)
    {
        var analysis = await _db.Analyses.FindAsync(id);
        if (analysis == null) return NotFound();
        _db.Analyses.Remove(analysis);
        await _db.SaveChangesAsync();
        return NoContent();
    }
}
```

### Calculation Endpoint

```csharp
[HttpPost("calculate/{analysisId}")]
public async Task<IActionResult> Calculate(int analysisId)
{
    // 1. Get current user
    var userId = User.FindFirstValue(ClaimTypes.NameIdentifier);

    // 2. Load analysis
    var analysis = await _db.Analyses.FindAsync(analysisId);
    if (analysis == null) return NotFound();

    // 3. Check authorization
    if (analysis.UserId != userId) return Forbid();

    // 4. Run calculation
    var model = new SchmertmannModel(analysis.Parameters);
    var result = await model.RunAsync();

    // 5. Save results
    analysis.Results = result;
    await _db.SaveChangesAsync();

    // 6. Return messaging contract response
    return Ok(new
    {
        status = "success",
        inputs = new List<string>(),
        warnings = result.Warnings,
        errors = new List<string>(),
        can_run = true,
        results = result,
        results_cleared = false
    });
}
```

## Entity Framework Patterns

### Entity with Relationships

```csharp
public class Analysis
{
    public int Id { get; set; }

    [Required]
    [MaxLength(255)]
    public string Name { get; set; }

    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;

    // Foreign key
    public int UserId { get; set; }

    // Navigation property
    public User User { get; set; }

    // JSONB column
    [Column(TypeName = "jsonb")]
    public JsonDocument Parameters { get; set; }

    // One-to-many relationship
    public ICollection<Layer> Layers { get; set; }
}
```

### DbContext Configuration

```csharp
public class ToolboxDbContext : DbContext
{
    public ToolboxDbContext(DbContextOptions<ToolboxDbContext> options)
        : base(options) { }

    public DbSet<Analysis> Analyses { get; set; }
    public DbSet<User> Users { get; set; }
    public DbSet<Layer> Layers { get; set; }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        // Configure relationships
        modelBuilder.Entity<Analysis>()
            .HasOne(a => a.User)
            .WithMany(u => u.Analyses)
            .HasForeignKey(a => a.UserId);

        // Configure JSONB
        modelBuilder.Entity<Analysis>()
            .Property(a => a.Parameters)
            .HasColumnType("jsonb");
    }
}
```

## Common Gotchas

### Async/Await

```csharp
// WRONG: Forgetting await
public async Task<IActionResult> Get()
{
    var items = _db.Analyses.ToListAsync(); // Missing await!
    return Ok(items); // Returns Task, not List
}

// CORRECT
public async Task<IActionResult> Get()
{
    var items = await _db.Analyses.ToListAsync();
    return Ok(items);
}
```

### Null Checking

```csharp
// WRONG: Assuming entity exists
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    return Ok(item.Name); // NullReferenceException if not found!
}

// CORRECT
public async Task<IActionResult> Get(int id)
{
    var item = await _db.Analyses.FindAsync(id);
    if (item == null) return NotFound();
    return Ok(item.Name);
}
```

### Saving Changes

```csharp
// WRONG: Forgetting to save
public async Task<IActionResult> Update(int id, UpdateDto dto)
{
    var item = await _db.Analyses.FindAsync(id);
    item.Name = dto.Name;
    // Missing SaveChangesAsync! Changes are lost.
    return Ok(item);
}

// CORRECT
public async Task<IActionResult> Update(int id, UpdateDto dto)
{
    var item = await _db.Analyses.FindAsync(id);
    item.Name = dto.Name;
    await _db.SaveChangesAsync();
    return Ok(item);
}
```
